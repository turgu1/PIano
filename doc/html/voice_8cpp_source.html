173

178 
179  while (frameCount--) {

182

184  int ipos = fipos - scaleBuffPos;
185  int idx;
186

189  if (getNormalFrames(scaleBuff, BUFFER_FRAME_COUNT) == 0) goto endLoop;
190  ipos -= BUFFER_FRAME_COUNT;

192  }
193 
194  old1 = *a++ = scaleBuff[idx = (ipos << 1)];
195  old2 = *a++ = scaleBuff[idx + 1];
196  }
197  else {

199  // the data already retrieved in the preceeding loop...
200  *a++ = old1;

206  if (++ipos >= BUFFER_FRAME_COUNT) {

208  ipos -= BUFFER_FRAME_COUNT;
209  scaleBuffPos += BUFFER_FRAME_COUNT;
210  }
211 
212  // We do a linear approximation... not the best, but good enough...
213  *b++ = scaleBuff[idx = (ipos << 1)];
214  *b++ = scaleBuff[idx + 1];

216  *c++ = diff;
