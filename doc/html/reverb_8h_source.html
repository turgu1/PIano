55  // Those are the ARM NEON SIMD vectorized version of PUT and GET.
56  // PUTv insure that if data is put after the end of the buffer space,
57  // it is copied back to the beginning of the array. The redundancy of
58  // data insure proper working conditions for vector based instructions
59  // to quickly get access to 4 data entries in a row.
60  //
61  // Here is the non-optimized data copy version:
62  //
63  /*
64  #define PUTv(ptr,v) vst1q_f32(ptr->tail, v); \
65  if (ptr->tail == ptr->buff) vst1q_f32(ptr->end, v); \
66  if ((ptr->tail += 4) >= ptr->end) { \
67  int k = ptr->tail - ptr->end; \
68  if (k > 0) memcpy(ptr->buff, ptr->tail - k, k << 2); \
69  ptr->tail = ptr->buff + k; \
70  }
71  */
72  // Here is the optimized version (memcpy replaced with SIMD instructions):
73 
74  inline void PUTv(fifop ptr, float32x4_t v) {
75  vst1q_f32(ptr->tail, v);
76  if (ptr->tail == ptr->buff) vst1q_f32(ptr->end, v);
77  if ((ptr->tail += 4) >= ptr->end) {
78  int k = ptr->tail - ptr->end;
79  if (k > 0)
80  vst1q_f32((ptr)->buff,
81  k == 1 ? vextq_f32(vld1q_f32(ptr->tail - 4),
82  vld1q_f32(ptr->buff + 1),
83  3) :
84  k == 2 ? vextq_f32(vld1q_f32(ptr->tail - 4),
85  vld1q_f32(ptr->buff + 2),
86  2) :
87  vextq_f32(vld1q_f32(ptr->tail - 4),
88  vld1q_f32(ptr->buff + 3),
89  1));
