97 
98  int channel = message->at(0) & MIDI_CHANNEL_MASK;
99 
100  unsigned char command = message->at(0) & ((unsigned char) MIDI_COMMAND_MASK);
101  unsigned char data1 = (count > 1) ? message->at(1) : 0;
102  unsigned char data2 = (count > 2) ? message->at(2) : 0;
103 
104  if (command == MIDI_SYSEX) {
105  command = message->at(0);
106  channel = -1;
107  }
108 
109  if (((command & 0xF0) == MIDI_SYSEX) ||
110  ((1 << channel) & midi->channelMask)) {
111 
112  switch (command) {
113  case MIDI_NOTE_ON:
114  midi->setNoteOn(data1 + cfg.midi.transpose, data2);
115  break;
116  case MIDI_NOTE_OFF:
117  midi->setNoteOff(data1 + cfg.midi.transpose, data2);
118  break;
119  case MIDI_CONTROL:
120  switch (data1) {
121  case 0x40:
122  if (midi->sustainIsOn() && (data2 < cfg.midi.sustainTreshold)) {
123  midi->setSustainOff();
124  }
125  else if (!midi->sustainIsOn() && (data2 >= cfg.midi.sustainTreshold)) {
126  midi->setSustainOn();
127  }
128  break;
129  case 0x47:
130  reverb->setRoomSize(0.7f + 0.29f * (data2 / 127.0f));
131  break;
132  case 0x4A:
133  masterVolume = data2 / 127.0f;
134  break;
135  default:
136  //logger.WARNING("Midi: Ignored Control: %02xh %d.\n",
137  // data1, data2);
138  break;
139  }
140  break;
141  case MIDI_PROGRAM:
142  if (!sound->holding()) {
143  sound->wait();
144  poly->inactivateAllVoices();
145  samples->loadNextLibrary();
146  sound->conti();
147  }
148  break;
149  default:
150  //logger.WARNING("Midi: Ignored Event: %02xh %d %d.\n",
151  // command, data1, data2);
152  break;
153  }
154  }
155

157

160  if (command == MIDI_SYSEX) {

163  }
